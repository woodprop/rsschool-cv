                        function anagrams(string $word, array $words): array {
                            $word_arr = str_split($word);
                            sort($word_arr);
                            $master = array_count_values($word_arr);
                            $res = [];

                            foreach ($words as $w) {
                                $arr = str_split($w);
                                sort($arr);
                                $current = array_count_values($arr);
                                if ($current == $master) $res[] = $w;
                            }

                        return $res;
                        }
